def insertion_sort(problema):

    for i in range(1, len(problema)):
        temp = problema[i]
        j = i - 1
        while (j >= 0 and temp < problema[j]):
            problema[j + 1] = problema[j]
            j = j - 1
        problema[j + 1] = temp
